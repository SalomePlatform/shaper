  FeaturesAPI
  ModelHighAPI
  ModelAPI

)

)

  SWIG_ADD_MODULE(FeaturesAPI python FeaturesAPI.i ${PROJECT_HEADERS})
ENDIF()
SWIG_LINK_LIBRARIES(FeaturesAPI ${SWIG_LINK_LIBRARIES})
